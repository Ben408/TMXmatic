57class TQEEngine: 

58 """ 

59 Translation Quality Estimation Engine. 

60  

61 Scores translation candidates using multiple metrics: 

62 - Accuracy (COMET, BERTScore, SBERT) 

63 - Fluency (LM perplexity) 

64 - Tone (SBERT similarity to exemplars) 

65 - Hallucination detection (UQLM) 

66 """ 

67 

68 def __init__(self, device: str = "cpu", lm_name: Optional[str] = None, 

69 sbert_name: Optional[str] = None, comet_ref_ckpt: Optional[str] = None, 

70 comet_qe_ckpt: Optional[str] = None, fp16: bool = False, 

71 comet_batch: int = 16): 

72 """ 

73 Initialize TQE Engine. 

74  

75 Args: 

76 device: Device to use ('cpu' or 'cuda') 

77 lm_name: HuggingFace model name for fluency scoring 

78 sbert_name: SentenceTransformer model name 

79 comet_ref_ckpt: Path to COMET reference-based checkpoint 

80 comet_qe_ckpt: Path to COMET-QE checkpoint 

81 fp16: Use half precision 

82 comet_batch: Batch size for COMET scoring 

83 """ 

84 self.device = device 

85 self.fp16 = fp16 

86 self.comet_batch = comet_batch 

87 

88 # Models (loaded on demand) 

89 self.lm_model = None 

90 self.lm_tokenizer = None 

91 self.sbert_model = None 

92 self.comet_model = None 

93 self.comet_qe_model = None 

94 

95 # Load models if names provided 

96 if TRANSFORMERS_AVAILABLE and lm_name: 

97 try: 

98 self.lm_tokenizer = AutoTokenizer.from_pretrained(lm_name) 

99 self.lm_model = AutoModelForCausalLM.from_pretrained(lm_name).to(self.device) 

100 if fp16 and self.device.startswith("cuda") and torch: 

101 try: 

102 self.lm_model.half() 

103 except Exception: 

104 pass 

105 logger.info(f"Loaded LM model: {lm_name}") 

106 except Exception as e: 

107 logger.warning(f"Could not load LM model {lm_name}: {e}") 

108 

109 if SBERT_AVAILABLE: 

110 model_name = sbert_name or "sentence-transformers/paraphrase-multilingual-mpnet-base-v2" 

111 try: 

112 self.sbert_model = SentenceTransformer(model_name) 

113 logger.info(f"Loaded SBERT model: {model_name}") 

114 except Exception as e: 

115 logger.warning(f"Could not load SBERT model: {e}") 

116 

117 if comet_ref_ckpt: 

118 try: 

119 self.comet_model = load_comet_model(comet_ref_ckpt, device=self.device, fp16=fp16) 

120 except Exception as e: 

121 logger.warning(f"Could not load COMET ref model: {e}") 

122 

123 if comet_qe_ckpt: 

124 try: 

125 self.comet_qe_model = load_comet_model(comet_qe_ckpt, device=self.device, fp16=fp16) 

126 except Exception as e: 

127 logger.warning(f"Could not load COMET-QE model: {e}") 

128 

129 def score_candidate(self, candidate: str, source: str, reference: Optional[str] = None, 

130 tone_exemplars: Optional[List[str]] = None, 

131 uqlm_check: bool = True) -> Dict: 

132 """ 

133 Score a single translation candidate. 

134  

135 Args: 

136 candidate: Candidate translation 

137 source: Source text 

138 reference: Optional reference translation 

139 tone_exemplars: Optional tone exemplar texts 

140 uqlm_check: Whether to check for hallucination 

141  

142 Returns: 

143 Dictionary with scores and metadata 

144 """ 

145 # UQLM hallucination check 

146 uq_result = {} 

147 uq_halluc = False 

148 if uqlm_check: 

149 try: 

150 uq_result = detect_hallucination(candidate, source=source) 

151 uq_halluc = uq_result.get('hallucination', False) or uq_result.get('is_hallucination', False) 

152 except Exception as e: 

153 logger.warning(f"UQLM check failed: {e}") 

154 

155 # Accuracy scoring 

156 acc_components = {} 

157 if reference and self.comet_model: 

158 try: 

159 raw = run_comet_ref_batch( 

160 self.comet_model, [source], [candidate], [reference], 

161 batch_size=1, device=self.device 

162 )[0] 

163 acc_components["comet_ref_raw"] = raw 

164 acc = normalize_comet_score(raw, raw_min=-1.0, raw_max=1.0) 

165 acc_components["comet_ref_norm"] = round(acc, 3) 

166 except Exception as e: 

167 logger.warning(f"COMET ref scoring failed: {e}") 

168 acc = compute_accuracy_with_reference_bertscore(candidate, reference) 

169 acc_components["bertscore"] = round(acc, 3) 

170 elif self.comet_qe_model and not reference: 

171 try: 

172 raw = run_comet_qe_batch( 

173 self.comet_qe_model, [source], [candidate], 

174 batch_size=1, device=self.device 

175 )[0] 

176 acc_components["comet_qe_raw"] = raw 

177 acc = normalize_comet_score(raw, raw_min=-1.0, raw_max=1.0) 

178 acc_components["comet_qe_norm"] = round(acc, 3) 

179 except Exception as e: 

180 logger.warning(f"COMET-QE scoring failed: {e}") 

181 acc = compute_accuracy_without_reference_sbert(source, candidate) 

182 acc_components["sbert_no_ref"] = round(acc, 3) 

183 else: 

184 if reference: 

185 acc = compute_accuracy_with_reference_bertscore(candidate, reference) 

186 acc_components["bertscore"] = round(acc, 3) 

187 else: 

188 acc = compute_accuracy_without_reference_sbert(source, candidate) 

189 acc_components["sbert_no_ref"] = round(acc, 3) 

190 

191 # UQLM penalty 

192 if uq_halluc: 

193 acc = acc * 0.25 

194 

195 # Fluency scoring 

196 if self.lm_model and self.lm_tokenizer: 

197 flu = compute_fluency_perplexity( 

198 candidate, self.lm_model, self.lm_tokenizer, self.device 

199 ) 

200 flu_model = getattr(self.lm_model.config, "_name_or_path", "causal-lm") 

201 else: 

202 flu = 50.0 

203 flu_model = "none" 

204 

205 # Tone scoring 

206 exemplars = tone_exemplars or [] 

207 if exemplars and self.sbert_model: 

208 tone = compute_tone_score(candidate, exemplars, self.sbert_model) 

209 else: 

210 tone = 50.0 

211 

212 # Aggregate scores 

213 weighted, decision = aggregate_scores(acc, flu, tone, uq_halluc) 

214 

215 return { 

216 "accuracy": round(float(acc), 2), 

217 "fluency": round(float(flu), 2), 

218 "tone": round(float(tone), 2), 

219 "weighted": round(float(weighted), 2), 

220 "decision": decision, 

221 "uq_hallucination": bool(uq_halluc), 

222 "per_metric_models": { 

223 "accuracy_components": acc_components, 

224 "fluency_model": flu_model, 

225 "tone_model": getattr(self.sbert_model, "name", "sbert") if self.sbert_model else "none", 

226 }, 

227 "uqlm_result": uq_result 

228 }

230 def score_xliff(self, xliff_path: str, candidates: Dict[str, List[str]], 

231 tmx_map: Optional[Dict[str, List[Dict]]] = None, 

232 uqlm_map: Optional[Dict[str, Dict]] = None, 

233 tone_exemplars_map: Optional[Dict[str, List[str]]] = None, 

234 out_path: str = "out.xlf", 

235 weights: Tuple[float, float, float] = (0.6, 0.25, 0.15), 

236 lang: str = "en", 

237 fuzzy_threshold: float = 0.8, 

238 fuzzy_enable: bool = True): 

239 """ 

240 Score translation candidates in XLIFF file. 

241  

242 Args: 

243 xliff_path: Path to input XLIFF file 

244 candidates: Dictionary mapping TU ID or source -> list of candidate translations 

245 tmx_map: Optional TMX translation memory map 

246 uqlm_map: Optional pre-computed UQLM results 

247 tone_exemplars_map: Optional tone exemplars by TU ID 

248 out_path: Output XLIFF path 

249 weights: Weights for (accuracy, fluency, tone) 

250 lang: Language code 

251 fuzzy_threshold: Fuzzy match threshold 

252 fuzzy_enable: Enable fuzzy TM matching 

253 """ 

254 tree, tu_list = parse_xliff(xliff_path) 

255 tmx_map = tmx_map or {} 

256 uqlm_map = uqlm_map or {} 

257 tone_exemplars_map = tone_exemplars_map or {} 

258 

259 for tu in tu_list: 

260 tu_id = tu["id"] 

261 src = tu["src"] 

262 existing_tgt = tu["tgt"] 

263 xml_node = tu["xml_node"] 

264 

265 # Find human reference (prefer TMX fuzzy match) 

266 human_ref = None 

267 if fuzzy_enable and tmx_map: 

268 match_result = fuzzy_tmx_lookup(src, tmx_map, threshold=fuzzy_threshold) 

269 if match_result: 

270 human_ref, similarity = match_result 

271 

272 if human_ref is None: 

273 # Use existing target if marked as human 

274 if existing_tgt and is_human_translation_from_xliff_tu(xml_node): 

275 human_ref = existing_tgt 

276 

277 # Get candidates 

278 tu_candidates = (candidates.get(tu_id) or candidates.get(src) or 

279 ([existing_tgt] if existing_tgt else [])) 

280 if not tu_candidates: 

281 continue 

282 

283 # Score all candidates 

284 best = None 

285 best_meta = None 

286 

287 for cand in tu_candidates: 

288 # Get UQLM result if available 

289 uq = uqlm_map.get(tu_id) or uqlm_map.get(src) or {} 

290 uq_halluc = bool(uq.get("hallucination") or uq.get("is_hallucination")) 

291 

292 # Get tone exemplars 

293 exemplars = tone_exemplars_map.get(tu_id) or tone_exemplars_map.get("default") or [] 

294 

295 # Score candidate 

296 meta = self.score_candidate( 

297 candidate=cand, 

298 source=src, 

299 reference=human_ref, 

300 tone_exemplars=exemplars, 

301 uqlm_check=not uq_halluc # Skip if already computed 

302 ) 

303 

304 # Override with pre-computed UQLM if available 

305 if uq_halluc: 

306 meta["uq_hallucination"] = True 

307 meta["accuracy"] = meta["accuracy"] * 0.25 

308 meta["weighted"], meta["decision"] = aggregate_scores( 

309 meta["accuracy"], meta["fluency"], meta["tone"], 

310 uq_halluc=True, weights=weights 

311 ) 

312 

313 # Choose best candidate 

314 if (best is None or 

315 meta["weighted"] > best_meta["weighted"] or 

316 (meta["decision"] == "accept_auto" and 

317 best_meta["decision"] != "accept_auto")): 

318 best = cand 

319 best_meta = meta 

320 

321 # Write best candidate to target 

322 target_node = tu["target_node"] 

323 if target_node is None: 

324 target_node = etree.SubElement(xml_node, "target") 

325 target_node.text = best 

326 

327 # Add TQE metadata 

328 props = {} 

329 acc_comps = best_meta["per_metric_models"].get("accuracy_components", {}) 

330 for k, v in acc_comps.items(): 

331 props[f"tqe:accuracy.{k}"] = str(v) 

332 props["tqe:accuracy"] = str(best_meta["accuracy"]) 

333 props["tqe:fluency"] = str(best_meta["fluency"]) 

334 props["tqe:tone"] = str(best_meta["tone"]) 

335 props["tqe:weighted_score"] = str(best_meta["weighted"]) 

336 props["tqe:decision"] = best_meta["decision"] 

337 props["tqe:uqm_hallucination"] = str(best_meta["uq_hallucination"]) 

338 props["tqe:models_used"] = json.dumps(best_meta["per_metric_models"]) 

339 props["tqe:timestamp"] = now_iso() 

340 

341 add_tqe_props_to_tu(xml_node, props) 

342 

343 write_xliff(tree, out_path) 

344 logger.info(f"Scored XLIFF written to {out_path}")
